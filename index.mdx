<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-2xl sm:text-3xl font-semibold text-gray-900 dark:text-white mb-3">
    Quick Start
  </h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">
    Get up and running with Revstack in minutes
  </p>
  
  <Steps>
    <Step title="Install the SDK">
      Install the Revstack SDK for your platform:
      
      <CodeGroup>
      ```bash npm
      npm install @revstackhq/node
      ```
      
      ```bash yarn
      yarn add @revstackhq/node
      ```
      
      ```bash pnpm
      pnpm add @revstackhq/node
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Initialize the client">
      Create a Revstack client instance with your secret key:
      
      ```typescript
      import { Revstack } from "@revstackhq/node";

      const revstack = new Revstack({
        secretKey: process.env.REVSTACK_SECRET_KEY
      });
      ```
      
      <Note>
        Get your secret key from the [Revstack Dashboard](https://app.revstack.dev)
      </Note>
    </Step>
    
    <Step title="Define your billing configuration">
      Create a `revstack.config.ts` file to define your plans and features:
      
      ```typescript revstack.config.ts
      import { defineConfig, defineFeature, definePlan } from "@revstackhq/core";

      const features = {
        "api-calls": defineFeature({
          key: "api-calls",
          name: "API Calls",
          type: "limit",
        }),
      };

      export default defineConfig({
        features,
        plans: [
          definePlan<typeof features>({
            key: "starter",
            name: "Starter",
            features: {
              "api-calls": { value_limit: 1000 },
            },
          }),
        ],
      });
      ```
    </Step>
    
    <Step title="Check entitlements">
      Use the SDK to check if a customer can access a feature:
      
      ```typescript
      const { allowed, remaining } = await revstack.entitlements.check(
        "customer_id",
        "api-calls"
      );

      if (!allowed) {
        throw new Error("API limit exceeded");
      }
      ```
      
      <Accordion title="Example Response">
        ```json
        {
          "allowed": true,
          "remaining": 750,
          "limit": 1000,
          "reset_at": "2024-02-01T00:00:00Z"
        }
        ```
      </Accordion>
    </Step>
  </Steps>
</div>
